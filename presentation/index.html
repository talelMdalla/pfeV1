  const payload = req.body;
  const evaluation = buildEvaluation(payload);
  const enriched = await enrichWithGroq(evaluation);

  // Sauvegarde MongoDB si disponible
  if (mongoose.connection.readyState === 1) {

      ...enriched
    });
  }

  res.json(enriched);
});

Modèle de données

5 modèles MongoDB

Schémas Mongoose structurés pour couvrir l'ensemble du cycle PFE.

👤

User

email, password (bcrypt), role (student/mentor/admin), profile { name, skills, level, interests }
📋

Evaluation

student (ref), profileLabel, overallScore, summary, aiRecommendation, strengths[], skills[], interests[]
📁

Project

title, description, student (ref), mentor (ref), status, progress, milestones [{ title, completed, dueDate }]
🎓

Mentor

user (ref), expertise[], availability, students[]
🏅

Certification

user (ref), title, skills[], issuedDate, issuer
RÉSILIENCE

Mode demo

Données de secours en constants/ permettant un fonctionnement complet sans connexion MongoDB.
